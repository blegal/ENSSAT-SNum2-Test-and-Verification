#include <stdio.h>
#include <stdlib.h>
#include <string>
#include <iostream>
#include <time.h>

int main( int argc, char **argv ){

	time_t seconds;
	time( &seconds );
	srand( (unsigned int) seconds);

	if( argc != 2 ){
		printf("Help :\n");
		printf("  ./FloatGenerator NumberOfData\n");
		printf("    NumberOfData = the number of float data you need.\n");
		return -1;
	}

	int n = atoi( argv[1] );

	for(int i=0; i<n; i++){
		float a = (float)rand();
		float b = (float)rand();
		float c = a / b;
		if( ((int)rand())%3 == 1 ){
			c = c * ((unsigned char)rand());
		}
		if( ((int)rand())%3 == 1 ){
			c = -c;
		}
		printf("%f\n", c);
	}
	
	return 1;
}
